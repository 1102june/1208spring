### 2. `NetworkModule.kt` (또는 `ApiClient.kt`)

```kotlin
import okhttp3.OkHttpClient
import okhttp3.logging.HttpLoggingInterceptor
import retrofit2.Retrofit
import retrofit2.converter.gson.GsonConverterFactory
import java.util.concurrent.TimeUnit

object NetworkModule {
    // Base URL 설정
    // 에뮬레이터: 10.0.2.2
    // 실제 기기: PC의 로컬 IP 주소 (예: 192.168.0.100)
    private const val BASE_URL = "http://10.0.2.2:8080/"  // 에뮬레이터용
    // private const val BASE_URL = "http://192.168.0.100:8080/"  // 실제 기기용
    
    private val loggingInterceptor = HttpLoggingInterceptor().apply {
        level = HttpLoggingInterceptor.Level.BODY
    }
    
    private val okHttpClient = OkHttpClient.Builder()
        .addInterceptor(loggingInterceptor)
        .connectTimeout(30, TimeUnit.SECONDS)
        .readTimeout(30, TimeUnit.SECONDS)
        .writeTimeout(30, TimeUnit.SECONDS)
        .build()
    
    private val retrofit = Retrofit.Builder()
        .baseUrl(BASE_URL)
        .client(okHttpClient)
        .addConverterFactory(GsonConverterFactory.create())
        .build()
    
    val apiService: ApiService = retrofit.create(ApiService::class.java)
}
```

```kotlin
import retrofit2.Response
import retrofit2.http.*

interface ApiService {
    
    // ========== 인증 ==========
    
    @POST("auth/login")

    // ========== OTP ==========
    
    @GET("auth/otp/email/check")
    suspend fun checkEmailDuplicate(@Query("email") email: String): Response<ApiResponse<Boolean>>
    
    @POST("auth/otp/send")
    suspend fun sendOtp(@Body request: OtpRequest): Response<ApiResponse<String>>
    
    @POST("auth/otp/verify")
    suspend fun verifyOtp(@Body request: OtpRequest): Response<ApiResponse<String>>
    
    @GET("auth/otp/status")
    suspend fun getOtpStatus(@Query("email") email: String): Response<ApiResponse<Boolean>>
    
    // ========== 주택 ==========
    
    @GET("api/housing/recommended")
    suspend fun getRecommendedHousing(
        @Header("X-User-Id") userId: String? = null,
        @Query("userId") userIdParam: String? = null,
        @Query("lat") lat: Double? = null,
        @Query("lon") lon: Double? = null,
        @Query("radius") radius: Int? = null,
        @Query("limit") limit: Int? = null
```

## 사용 예시

### `HousingRepository.kt`

```kotlin
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

class HousingRepository {
    private val apiService = NetworkModule.apiService
    
    suspend fun getActiveHousing(userId: String): Result<List<HousingResponse>> {
        return withContext(Dispatchers.IO) {
            try {
                val response = apiService.getActiveHousing(
                    userId = userId,
                    userIdParam = userId
                )
                
                if (response.isSuccessful && response.body()?.success == true) {
                    Result.success(response.body()!!.data ?: emptyList())
                } else {
                    Result.failure(Exception(response.body()?.message ?: "Unknown error"))
                }
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
    
    suspend fun getRecommendedHousing(
        userId: String,
        lat: Double? = null,
        lon: Double? = null,
        radius: Int = 5000,
        limit: Int = 10
    ): Result<List<HousingResponse>> {
        return withContext(Dispatchers.IO) {
            try {
                val response = apiService.getRecommendedHousing(
                    userId = userId,
                    userIdParam = userId,
                    lat = lat,
                    lon = lon,
                    radius = radius,
                    limit = limit
                )
                
                if (response.isSuccessful && response.body()?.success == true) {
                    Result.success(response.body()!!.data ?: emptyList())
                } else {
                    Result.failure(Exception(response.body()?.message ?: "Unknown error"))
                }
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
}
```

### `ViewModel.kt`

```kotlin
import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.launch

class HousingViewModel : ViewModel() {
    private val repository = HousingRepository()
    
    private val _housingList = MutableStateFlow<List<HousingResponse>>(emptyList())
    val housingList: StateFlow<List<HousingResponse>> = _housingList
    
    private val _isLoading = MutableStateFlow(false)
    val isLoading: StateFlow<Boolean> = _isLoading
    
    private val _error = MutableStateFlow<String?>(null)
    val error: StateFlow<String?> = _error
    
    fun loadActiveHousing(userId: String) {
        viewModelScope.launch {
            _isLoading.value = true
            _error.value = null
            
            repository.getActiveHousing(userId)
                .onSuccess { housingList ->
                    _housingList.value = housingList
                }
                .onFailure { exception ->
                    _error.value = exception.message
                }
            
            _isLoading.value = false
        }
    }
}
```

### `Activity/Fragment`에서 사용

```kotlin
class HousingActivity : AppCompatActivity() {
    private lateinit var viewModel: HousingViewModel
    
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        viewModel = ViewModelProvider(this)[HousingViewModel::class.java]
        
        // 데이터 관찰
        lifecycleScope.launch {
            viewModel.housingList.collect { housingList ->
                // UI 업데이트
                updateHousingList(housingList)
            }
        }
        
        lifecycleScope.launch {
            viewModel.isLoading.collect { isLoading ->
                // 로딩 상태 표시
                progressBar.visibility = if (isLoading) View.VISIBLE else View.GONE
            }
        }
        
        lifecycleScope.launch {
            viewModel.error.collect { error ->
                error?.let {
                    // 에러 메시지 표시
                    Toast.makeText(this@HousingActivity, it, Toast.LENGTH_SHORT).show()
                }
            }
        }
        
        // 데이터 로드
        val userId = getUserId() // SharedPreferences 등에서 가져오기
        viewModel.loadActiveHousing(userId)
    }
}
```

## 에러 처리

### `ApiException.kt`

```kotlin
sealed class ApiException(message: String) : Exception(message) {
    class NetworkException(message: String) : ApiException(message)
    class HttpException(val code: Int, message: String) : ApiException(message)
    class UnknownException(message: String) : ApiException(message)
}
```

### 에러 처리 개선된 Repository

```kotlin
class HousingRepository {
    private val apiService = NetworkModule.apiService
    
    suspend fun getActiveHousing(userId: String): Result<List<HousingResponse>> {
        return withContext(Dispatchers.IO) {
            try {
                val response = apiService.getActiveHousing(
                    userId = userId,
                    userIdParam = userId
                )
                
                when {
                    response.isSuccessful && response.body()?.success == true -> {
                        Result.success(response.body()!!.data ?: emptyList())
                    }
                    response.code() == 400 -> {
                        Result.failure(ApiException.HttpException(400, "잘못된 요청입니다."))
                    }
                    response.code() == 404 -> {
                        Result.failure(ApiException.HttpException(404, "데이터를 찾을 수 없습니다."))
                    }
                    response.code() == 500 -> {
                        Result.failure(ApiException.HttpException(500, "서버 오류가 발생했습니다."))
                    }
                    else -> {
                        Result.failure(ApiException.UnknownException(response.body()?.message ?: "Unknown error"))
                    }
                }
            } catch (e: java.net.UnknownHostException) {
                Result.failure(ApiException.NetworkException("서버에 연결할 수 없습니다. 네트워크를 확인해주세요."))
            } catch (e: java.net.SocketTimeoutException) {
                Result.failure(ApiException.NetworkException("요청 시간이 초과되었습니다."))
            } catch (e: Exception) {
                Result.failure(ApiException.UnknownException(e.message ?: "알 수 없는 오류가 발생했습니다."))
            }
        }
    }
}
```

## 🔍 디버깅 팁

1. **OkHttp 로깅**: `HttpLoggingInterceptor`를 사용하여 요청/응답 로그 확인
2. **서버 로그 확인**: Spring Boot 콘솔에서 SQL 쿼리 및 에러 확인
3. **네트워크 상태 확인**: 실제 기기에서 테스트할 때는 PC와 같은 Wi-Fi에 연결되어 있는지 확인
4. **Base URL 확인**: 에뮬레이터는 `10.0.2.2`, 실제 기기는 PC의 로컬 IP 주소 사용

## 📝 참고사항

- 모든 API 호출은 `suspend` 함수로 정의하여 Coroutines 사용
- `Result` 타입을 사용하여 성공/실패 처리
- ViewModel에서 StateFlow를 사용하여 UI 상태 관리
- 에러 처리는 사용자 친화적인 메시지로 변환하여 표시
